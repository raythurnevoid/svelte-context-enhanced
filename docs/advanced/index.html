Svelte Typed Context

Advanced Usage

You can create your own context typings and implementations using the basic key interface ContextKey<T> and the functions setContext and getContext.

AdvancedExample.svelte 
<script lang="ts">
	import AdvancedExampleChild from "./AdvancedExampleChild.svelte";
	import { setContext } from "@raythurnevoid/svelte-context-enhanced";
	import { contextKey } from "./AdvancedExampleContext";

	setContext(contextKey, "value");
	// setContext(contextKey, 123); // This will fail type checking
</script>

<script lang="ts">
	import { getContext } from "@raythurnevoid/svelte-context-enhanced";
	import { contextKey } from "./AdvancedExampleContext";

	const value = getContext(contextKey); // inferred type is string
</script>

API

createContext takes no arguments but wants you to set it's generic type in order to type the given context setter and getter. 

/**
 * Credits: https://github.com/KamenKolev/svelte-typed-context
 */
/**
 * Provided as key to getContext and setContext in order to enable strict typing
 */
export interface ContextKey<T = unknown> {
}
declare type getContext = <T>(key: ContextKey<T>) => undefined | T;
declare type setContext = <T>(key: ContextKey<T>, context: T) => void;
export declare const getContext: getContext;
export declare const setContext: setContext;
export {};
