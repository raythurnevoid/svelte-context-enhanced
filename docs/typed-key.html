Svelte Typed Context

Typed Key

With the typed ContextKey you can keep the default svelte's API while keeping type checking

Usage

Use ContextKey with the setContext and getContext methods provided by svelte-context-enhanced.

TypedKeyExample.svelte 
<script lang="ts" context="module">
	import type { ContextKey } from '@raythurnevoid/svelte-context-enhanced';

	export const contextKey: ContextKey<string> = {};
</script>

<script lang="ts">
	import AdvancedExampleChild from './TypedKeyExampleChild.svelte';
	import { setContext } from '@raythurnevoid/svelte-context-enhanced';

	setContext(contextKey, 'value with typed key');
	// setContext(contextKey, 123); // This will fail type checking
</script>

<AdvancedExampleChild />
TypedKeyExampleChild.svelte 
<script lang="ts">
	import { getContext } from '@raythurnevoid/svelte-context-enhanced';
	import { contextKey } from './TypedKeyExample.svelte';

	const value = getContext(contextKey); // inferred type is string
</script>
